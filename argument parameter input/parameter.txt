$0 is the name of the command
$1 first argument
$2 second argument
$3 third argument etc. etc
$# total number of arguments. A number (For example: 5).
$@ all the arguments will be listed (For example: login.sh , user, password).

sh-3.2$ cat a.sh
echo $#  #print the number of cmd line args.
sh-3.2$ ./a.sh
0
sh-3.2$ ./a.sh foo
1
sh-3.2$ ./a.sh foo bar
2
sh-3.2$ ./a.sh foo bar baz
3
